module Languages.MiniML.Programs where

open import Signature
open import MonadicFragment
open import Languages.MiniML

open import Fragments.Nat    using (`nat)
open import Fragments.Bools  using (`bool)
open import Fragments.Lambda using (`fun)

open import Data.List
open import Data.List.Relation.Unary.All
open import Data.List.Membership.Propositional
open import Data.List.Relation.Unary.Any
open import Data.Nat
open import Data.Bool
open import Data.Product
open import Data.Maybe

open import Relation.Binary.PropositionalEquality using (_≡_ ; refl)

module _ where

  -- if (true) then (4) else (1 + 2)
  prog₀ : Iμ (Syntax miniml) ([] , `nat)
  prog₀ = `ite (`bval true) (`nval 4) (`add (`nval 1) (`nval 2))

  -- prog₀ ~~> 4
  test₀ : execMiniML prog₀ 10 [] ≡ just 4
  test₀ = refl


  -- (λ x → ncase x { Z → Z ; (S n) → n }) 5
  prog₁ : Iμ (Syntax miniml) ([] , `nat)
  prog₁ =
    `app
      (`abs (`ncase (`var (here refl)) (`nval 0) (`var (here refl))))
      (`nval 5)

  test₁ : execMiniML prog₁ 10 [] ≡ just 4
  test₁ = refl


  -- Multiplication 
  --
  -- ((fix μ x → ncase x { Z → (λ y → Z) ; (S n) → (λ y → y + μ n y)}) 8) 8
  --
  -- ≈ 8 * 8  
  prog₂ : Iμ (Syntax miniml) ([] , `nat)
  prog₂ = `app (`app
    (`fix
      (`ncase
        (`var (here refl)) -- x
        (`abs (`nval 0))
        (`abs
          (`add
            (`var (here refl)) -- y 
              (`app
                (`app
                  (`var (there (there (there (here refl))))) -- μ 
                  (`var (there (here refl))))                -- n
                (`var (here refl)))))))                      -- y 
    (`nval 8)) (`nval 8)

  -- prog₂ ~~> 64
  test₂ : execMiniML prog₂ 10 [] ≡ just 64
  test₂ = refl


  -- Factorial
  --
  -- (fix μ x → ncase x { Z → 1
  --                    ; (S n) → ((fix μ' y → ncase y
  --                                             { Z → (λ z → Z)
  --                                             ; (S m) → (λ z → z + μ' m z)
  --                                             }) x)
  --                    } (μ n)) 6
  --
  -- ≈ 6!
  prog₃ : Iμ (Syntax miniml) ([] , `nat)
  prog₃ =
    `app
      (`fix
        (`ncase
          (`var (here refl)) -- x
          (`nval 1)
          (`app
           (`app
             (`fix
               (`ncase
                 (`var (here refl))
                 (`abs (`nval 0))
                 (`abs
                   (`add
                     (`var (here refl))    -- z
                     (`app
                       (`app
                         (`var (there (there (there (here refl)))))
                         (`var (there (here refl))))
                       (`var (here refl))) -- z 
               ))))
             (`var (there (here refl)))) -- x
           (`app
             (`var (there (there (here refl))))   -- μ 
             (`var (here refl))                   -- n 
        ))))
      (`nval 6)

  -- prog₃ ~~> 720
  test₃ : execMiniML prog₃ 10 [] ≡ just 720
  test₃ = refl
