{-# OPTIONS --safe --without-K #-}

module Fragments.Bools where

open import Data.Bool
open import Data.Product
open import Data.Vec
open import Data.List
open import Data.List.Relation.Unary.All

open import Canon
open import Fragment
open import Signature

open import Function
import Level as L

module _ where

  data BoolTyShape : Set where bool : BoolTyShape
  BoolTyΣ = BoolTyShape ▹ const 0

  `bool : ⦃ BoolTyΣ ≼ σ ⦄ → μ σ
  `bool = inject (bool , L.lift [])

  BoolVal : Algebra BoolTyΣ Set
  alg BoolVal (bool , L.lift []) = Bool

  data BoolExprShape {σ} ⦃ _ : BoolTyΣ ≼ σ ⦄ : μ σ → Set where
    bval : Bool → BoolExprShape `bool
    ite  : ∀ {t} → BoolExprShape t

  BoolExprΣ : ⦃ _ : BoolTyΣ ≼ σ ⦄ → ISignature _ (μ σ)
  BoolExprΣ = BoolExprShape ▸ λ where
    (bval b)  → []
    (ite {t}) → `bool ∷ t ∷ t ∷ [] 

  open IAlgebra
  interpBoolAlg : ⦃ _ : BoolVal ⊆ V ⦄ → IAlgebra BoolExprΣ (fold V)
  ialg interpBoolAlg (bval b , _                   ) = ↑ b
  ialg interpBoolAlg (ite , L.lift (c ∷ t ∷ e ∷ [])) = if ↓ c then t else e

  frag = mkfragment interpBoolAlg
