module _ where

  open Fragment {η = η}
  open Domains
  open Open
  open ValueType 

  open Union
  
  data `Arrow (X : Set) : Set where
    _⇒_ : (s t : X) → `Arrow X 

  LamExpr : ExpressionType `Arrow Ctx
  closer LamExpr (_ , σ) R (Γ , t)
         = t ∈ Γ                                              -- var
         ⊎ (∃ λ s → R (Γ , inja σ (s ⇒ t)) × R (Γ , s))       -- app
         ⊎ (∃₂ λ s t' → t ≡ inja σ (s ⇒ t') × R (s ∷ Γ , t')) -- abs

  LamVal : ValueType `Arrow
  value LamVal R (s ⇒ t) = R s → R t

  LamDom : Domains Ctx
  ty   LamDom = `Arrow
  expr LamDom = LamExpr
  val  LamDom = LamVal

module _ where

  Env : {X : Set} → (V : ISet X) → Ctx X → Set
  Env V Γ = All V Γ

  open Effects

  record Reader (η : Effects Ctx) : Set₁ where
  
    field ask : ∀ {X} {V : ISet X}{Γ t}
                → M η V (Env V ∘ proj₁) (Γ , t)
                
          abs : ∀ {X} {V : ISet X}{P}{Γ s t}
                → M η V P (s ∷ Γ , t)
                → (ξ : `Arrow X ≤ X)
                → M η V P (Γ , inja (proj₂ ξ) (s ⇒ t)) 

module _ {η : Effects Ctx} where

  typeof : ∀ {X : Set} {t : X} {Γ : List X} → t ∈ Γ → X
  typeof {t = t} _ = t
  
  open Monadic ⦃...⦄
  open Reader  ⦃...⦄
  open Inverse renaming (f to ↑ ; f⁻¹ to ↓)

  evalLam : ⦃ Monadic η ⦄ → ⦃ Reader η ⦄ → `Arrow ⊢ LamExpr ⟨ η ⟩↦ LamVal
  run evalLam ξ μ (inj₁ t∈Γ) = do
    Γ ← ask {t = typeof t∈Γ}
    pure (lookup Γ t∈Γ)
  run evalLam ξ μ (inj₂ (inj₁ (_ , ff , fx))) = do
    f ← ff
    x ← fx
    pure (↓ μ f x)
  run evalLam ξ μ (inj₂ (inj₂ (_ , _ , refl , f))) = abs f ξ

  LamFrag : ⦃ Monadic η ⦄ → ⦃ Reader η ⦄ → Fragment LamDom
  Fragment.denote LamFrag = evalLam
