{-# OPTIONS --safe --without-K #-}

open import Composable.Algebra

module Composable.Fragments.Pair (K : Set → Set) (η : Effects K) where 

open import Composable.Transformer
open import Composable.Data.Core
open import Composable.Data.Instances
open import Composable.Fragments.Core

open import Data.Product
open import Data.Empty
open import Data.Sum hiding (swap)
open import Data.Unit

open import Relation.Ternary.Construct.Sets.Union
open import Relation.Binary.PropositionalEquality

open import Function

module _ where

  open Fragment {η = η}
  open Domains
  open Open
  open ValueType
  open Union

  data `Pair (X : Set) : Set where
    pair : X → X → `Pair X

  PairExpr : ExpressionType `Pair K
  closer PairExpr (_ , σ) R (κ , t)
         = (∃₂ λ t₁ t₂ → t ≡ inja σ (pair t₁ t₂) × R (κ , t₁) × R (κ , t₂))     -- mkpair
         ⊎ (∃₂ λ t₁ t₂ → t ≡ inja σ (pair t₁ t₂) × R (κ , inja σ (pair t₂ t₁))) -- swap 
         ⊎ (∃ λ t' → R (κ , inja σ (pair t t')))                                -- fst

  PairVal : ValueType `Pair
  value PairVal R (pair t₁ t₂) = R t₁ × R t₂

  PairDom : Domains K
  ty   PairDom = `Pair
  expr PairDom = PairExpr
  val  PairDom = PairVal

module _ where
 
  open Monadic ⦃...⦄
  open Inverse renaming (f to ↑ ; f⁻¹ to ↓) 
  
  evalPair : ⦃ Monadic η ⦄ → `Pair ⊢ PairExpr ⟨ η ⟩↦ PairVal
  run evalPair ξ μ (inj₁ (_ , _ , refl , fx , fy)) = do
    x ← fx
    y ← fy
    pure (↑ μ (x , y))
  run evalPair ξ μ (inj₂ (inj₁ (_ , _ , refl , fxy))) = do
    xy ← fxy
    pure (↑ μ (swap (↓ μ xy)))
  run evalPair ξ μ (inj₂ (inj₂ (_ , fxy))) = do
    xy ← fxy
    pure (proj₁ (↓ μ xy))

  PairFrag : ⦃ Monadic η ⦄ → Fragment PairDom
  Fragment.denote PairFrag = evalPair
