{-# OPTIONS --safe --without-K #-}

module Fragments.Nat where 

open import Data.Nat
open import Data.Product
open import Data.Vec
open import Data.List
open import Data.List.Relation.Unary.All

open import Canon
open import Fragment
open import Signature

open import Relation.Unary using (IUniversal ; _⇒_ ; _⊢_)
open import Function
import Level as L

module _ where

  --
  -- Natural number types and values
  --

  data NatTyShape : Set where nat : NatTyShape
  NatTyΣ = NatTyShape ▹ const 0

  `nat : ⦃ NatTyΣ ≼ σ ⦄ → μ σ 
  `nat = inject (nat , L.lift [])

  NatVal : Algebra NatTyΣ Set
  alg NatVal (nat , L.lift []) = ℕ

  --
  -- Arithmetic expressions, and their interpreters
  --

  data ArithExprShape {σ} ⦃ _ : NatTyΣ ≼ σ ⦄ : μ σ → Set where
    nval : ℕ → ArithExprShape `nat
    add  :     ArithExprShape `nat

  ArithExprΣ : ⦃ _ : NatTyΣ ≼ σ ⦄ → ISignature _ (μ σ)
  ArithExprΣ = ArithExprShape ▸ λ where
    (nval _) → []
    add      → `nat ∷ `nat ∷ []     

  open IAlgebra
  interpArithAlg : ⦃ _ : NatVal ⊆ V ⦄ → IAlgebra ArithExprΣ (fold V)
  ialg interpArithAlg (nval n , _                    ) = ↑ n
  ialg interpArithAlg (add    , L.lift (n ∷ m ∷ [])) = ↑ (↓ n + ↓ m)

  frag = mkfragment interpArithAlg
