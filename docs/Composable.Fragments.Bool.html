{-# OPTIONS --safe --without-K #-}

open import Composable.Algebra

module Composable.Fragments.Bool (K : Set → Set) (η : Effects K) where 

open import Composable.Transformer
open import Composable.Data.Core
open import Composable.Data.Instances
open import Composable.Fragments.Core

open import Data.Product
open import Data.Empty
open import Data.Sum
open import Data.Unit
open import Data.Bool

open import Relation.Ternary.Construct.Sets.Union
open import Relation.Binary.PropositionalEquality

open import Function

module _ where

  open Fragment {η = η}
  open Domains
  open Open
  open ValueType

  open Union

  data `Bool (X : Set) : Set where
    bool : `Bool X

  BoolExpr : ExpressionType `Bool K
  closer BoolExpr (_ , σ) R (κ , t)
         = t ≡ inja σ bool                             -- false
         ⊎ t ≡ inja σ bool                             -- true
         ⊎ R (κ , inja σ bool) × R (κ , t) × R (κ , t) -- if then else

  BoolVal : ValueType `Bool
  value BoolVal R bool = Bool

  BoolDom : Domains K
  ty   BoolDom = `Bool
  expr BoolDom = BoolExpr
  val  BoolDom = BoolVal

module _ where

  open Monadic ⦃...⦄
  open Inverse renaming (f to ↑ ; f⁻¹ to ↓)

  evalBool : ⦃ Monadic η ⦄ → `Bool ⊢ BoolExpr ⟨ η ⟩↦ BoolVal
  run evalBool ξ μ (inj₁ refl)                     = pure (↑ μ false)
  run evalBool ξ μ (inj₂ (inj₁ refl))              = pure (↑ μ true)
  run evalBool ξ μ (inj₂ (inj₂ (fe₁ , fe₂ , fe₃))) = do
    e₁ ← fe₁
    e₂ ← fe₂
    e₃ ← fe₃
    pure (if ↓ μ e₁ then e₂ else e₃)

  BoolFrag : ⦃ Monadic η ⦄ → Fragment BoolDom
  Fragment.denote BoolFrag = evalBool
