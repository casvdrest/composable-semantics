module _ where

  IndexType : Set₁
  IndexType = Set → Set

  _≋ᵀ_ : Rel IndexType (suc 0ℓ)
  T₁ ≋ᵀ T₂ = ∀ {X} → T₁ X ↔ T₂ X

module _ where 
  
  record ValueType (T : Set → Set) : Set₁ where 
    field value : ∀ {X} → X ▸ T X

  open ValueType

  Mono : ∀ {T X} → T X ≤ X → ValueType T → ISet X → Set
  Mono (_ , σ) V R = ∀ {t} → value V R t ↔ R (inja σ t)

  _⊙⟨_⟩v_ : ∀ {I J K} → ValueType I → I ∙ J ≣ K → ValueType J → ValueType K
  value (D ⊙⟨ σ ⟩v E) R t with from (σ _) t
  ... | this  i   = value D R i
  ... | that  j   = value E R j
  ... | these i j = value D R i -- × value D R i ↔ value E R j

  _⊙⟨_⟩v_≣_ : ValueType T₁ → T₁ ∙ T₂ ≣ T₃ → ValueType T₂ → ValueType T₃ → Set₁
  V₁ ⊙⟨ σ ⟩v V₂ ≣ V₃ = ∀ {X W t} → value (V₁ ⊙⟨ σ ⟩v V₂) {X} W t ↔ value V₃ W t

  _≈ᵛ_ : (V₁ V₂ : ValueType T) → Set₁
  _≈ᵛ_ {T} V₁ V₂ = ∀ {X} {W} {t : T X} → value V₁ W t ↔ value V₂ W t

-- Interchangeability

module _ where 

  open ValueType

  Interchangeable : (σ : T₁ ∙ T₂ ≣ T₃) → (V₁ : ValueType T₁) → (V₂ : ValueType T₂) → (X : Set) → (W : ISet X) → T₃ X → Set
  Interchangeable σ V₁ V₂ X W t = U.From⟪ U , U , (λ (t₁ , t₂) → value V₁ W t₁ ↔ value V₂ W t₂) ⟫ t  
    where module U = Union (σ X)
    
  record _←⟨_⟩→_ (V₁ : ValueType T₁) (σ : T₁ ∙ T₂ ≣ T₃) (V₂ : ValueType T₂) : Set₁ where
    field intr : ∀ {X : Set} {W t}
                 → Interchangeable σ V₁ V₂ X W t 


  ←⟨⟩→-copy : ∀ {V : ValueType T} → V ←⟨ (λ _ → ∙-copy _) ⟩→ V
  _←⟨_⟩→_.intr ←⟨⟩→-copy = id-↔ _

  ←⟨⟩→-disjoint : ∀ {V₁ : ValueType T₁} {V₂ : ValueType T₂}
                  → V₁ ←⟨ (λ _ → ∙-disjoint) ⟩→ V₂
  _←⟨_⟩→_.intr ←⟨⟩→-disjoint {t = t} with from-inv' ∙-disjoint t
  ... | From.this a i refl = tt
  ... | From.that b i refl = tt
  ... | From.these a b i (refl , snd) = tt

module _ where
  
  ExpressionType : (Set → Set) → (Set → Set) → Set₁
  ExpressionType T F = Open (λ X → Data (F X × X)) T

  _⊙⟨_⟩e_≣_ : ExpressionType T₁ F → T₁ ∙ T₂ ≣ T₃ → ExpressionType T₂ F → ExpressionType T₃ F → Set₁
  E₁ ⊙⟨ σt ⟩e E₂ ≣ E₃ =
    ∀ {X} {ξ : _ ≤ X} {W t}
    → ⟪ E₁ ▿ (_ , σt) ⟫ ξ W t ∙ ⟪ (E₂ ▿ (_ , ∙-comm σt)) ⟫ ξ W t ≣ ⟪ E₃ ⟫ ξ W t

  _⊙⟨_⟩e_ : ExpressionType T₁ F → T₁ ∙ T₂ ≣ T₃ → ExpressionType T₂ F → ExpressionType T₃ F
  closer (E₁ ⊙⟨ σ ⟩e E₂) ξ R = ⟪ E₁ ⟫ (≤-trans (_ , σ _         ) ξ) R
                             ∪ ⟪ E₂ ⟫ (≤-trans (_ , ∙-comm (σ _)) ξ) R 

  _⟨_⟩≋ᴱ_ : ExpressionType T₁ F → T₁ ≋ᵀ T₂ → ExpressionType T₂ F → Set₁
  E₁ ⟨ eq ⟩≋ᴱ E₂ = ∀ {X W} {ξ : _ ≤ X} {t} → ⟪ E₁ ⟫ ξ W t ↔ ⟪ E₂ ⟫ (proj₁ ξ , coe ⦃ ∙-respects-≈ˡ ⦄ eq (proj₂ ξ)) W t

module _ where

  record Domains (F : Set → Set) : Set₁ where
    field
      ty     : IndexType
      expr   : ExpressionType ty F
      val    : ValueType ty
  open Domains

  composeDomains : (Δ₁ Δ₂ : Domains F) → (∃ λ T → ty Δ₁ ∙ ty Δ₂ ≣ T) → Domains F
  ty   (composeDomains Δ₁ Δ₂ (T , _)) = T
  expr (composeDomains Δ₁ Δ₂ (_ , σ)) = expr Δ₁ ⊙⟨ σ ⟩e expr Δ₂
  val  (composeDomains Δ₁ Δ₂ (_ , σ)) = val  Δ₁ ⊙⟨ σ ⟩v val  Δ₂

  infixr 9 composeDomains
  syntax composeDomains Δ₁ Δ₂ σ = Δ₁ ⊙⟨ σ ⟩d Δ₂

  record _⊙_≣_ {F} (Δ₁ Δ₂ Δ₃ : Domains F) : Set₁ where
    constructor _∣_∣_▹_
    field
      σt : ty   Δ₁ ∙        ty   Δ₂ ≣ ty   Δ₃
      σe : expr Δ₁ ⊙⟨ σt ⟩e expr Δ₂ ≣ expr Δ₃
      σv : val  Δ₁ ⊙⟨ σt ⟩v val  Δ₂ ≣ val  Δ₃
      ι  : val Δ₁ ←⟨ σt ⟩→ val Δ₂

  instance ⊙-rel : ∀ {F} → Rel₃ (Domains F)
  (⊙-rel Rel₃.∙ Δ₁ ≣ Δ₂) Δ₃ = Δ₁ ⊙ Δ₂ ≣ Δ₃
