-- Syntactic index, expression and value types
module _ where

  open ValueType

  ⟦_⟧⋆ : ⊤ ▷ ⊤ → Set → Set
  ⟦ c ⟧⋆ x = ⟦ c ⟧ (λ where tt → x) tt

  -- Separations on ⊤-indexed containers
  instance ⊤▷⊤-rel : Rel₃ (⊤ ▷ ⊤)
  Rel₃._∙_≣_ ⊤▷⊤-rel c₁ c₂ c₃ = ⟦ c₁ ⟧⋆ ∙ ⟦ c₂ ⟧⋆ ≣ ⟦ c₃ ⟧⋆

  ExpressionTypeˢ : (T : ⊤ ▷ ⊤) → (F : Set → Set) → Set₁
  ExpressionTypeˢ T F = Open (λ X → (F X × X) ▷ (F X × X)) ⟦ T ⟧⋆

  record ValueTypeˢ (T : ⊤ ▷ ⊤) : Set₁ where
    field valueˢ : ∀ {X} → X ▷ ⟦ T ⟧⋆ X

  open ValueTypeˢ 

  -- ValueTypeˢ : (T : ⊤ ▷ ⊤) → Set₁
  -- ValueTypeˢ T = Open (λ X → X ▷ ⟦ T ⟧⋆ X) ⟦ T ⟧⋆

  ∥_∥e : ∀ {T} → ExpressionTypeˢ T F → ExpressionType ⟦ T ⟧⋆ F
  closer ∥ e ∥e ξ = ⟦ closer e ξ ⟧

  ∥_∥v : ∀ {T} → ValueTypeˢ T → ValueType ⟦ T ⟧⋆ 
  value ∥ v ∥v = ⟦ valueˢ v ⟧


-- Syntactic domains and fragments
module _ where 


  record SDomains (F : Set → Set) : Set₁ where
    field
      tyˢ   : ⊤ ▷ ⊤
      exprˢ : ExpressionTypeˢ tyˢ F 
      valˢ  : ValueTypeˢ tyˢ

  open Open

  record SFragment {η} (Δ : SDomains F) : Set₁ where 
    field sdenote : let open SDomains Δ in
                    ⟦ tyˢ ⟧⋆ ⊢ ∥ exprˢ ∥e ⟨ η ⟩↦ ∥ valˢ ∥v


module _ {η : Effects F} where 

  open Domains
  open SDomains

  ∥_∥d : SDomains F → Domains F
  ty   ∥ Δ ∥d = ⟦ tyˢ   Δ ⟧⋆
  expr ∥ Δ ∥d = ∥ exprˢ Δ ∥e
  val  ∥ Δ ∥d = ∥ valˢ  Δ ∥v

  open Fragment
  open SFragment

  ∥_∥f : ∀ {Δ} → SFragment {η = η} Δ → Fragment {η = η} ∥ Δ ∥d 
  run (denote ∥ f ∥f) = run (sdenote f)

module _ where

  open ValueTypeˢ

  -- Heterogeneous composition for syntactic value types 
  _⊙⟪_⟫v_ : ∀ {I J K}
           → ValueTypeˢ I → I ∙ J ≣ K → ValueTypeˢ J
           → ValueTypeˢ K 
  Command (valueˢ (V₁ ⊙⟪ σ ⟫v V₂)) o with from-inv' (σ _) o
  ... | From.this  a   i P.refl        = Command (valueˢ V₁) a
  ... | From.that  b   i P.refl        = Command (valueˢ V₂) b
  ... | From.these a b i (P.refl , rx) = Command (valueˢ V₁) a
  Response (valueˢ (V₁ ⊙⟪ σ ⟫v V₂)) {o} c with from-inv' (σ _) o
  ... | From.this  a   i P.refl         = Response (valueˢ V₁) c
  ... | From.that  b   i P.refl         = Response (valueˢ V₂) c
  ... | From.these a b i (P.refl , snd) = Response (valueˢ V₁) c
  next (valueˢ (V₁ ⊙⟪ σ ⟫v V₂)) {o} c r with from-inv' (σ _) o
  ... | From.this  a   i P.refl         = next (valueˢ V₁) c r
  ... | From.that  b   i P.refl         = next (valueˢ V₂) c r
  ... | From.these a b i (P.refl , snd) = next (valueˢ V₁) c r

module _ where

  _⊙⟪_⟫e_ : ∀ {I J K}
            → ExpressionTypeˢ I F → I ∙ J ≣ K → ExpressionTypeˢ J F
            → ExpressionTypeˢ K F
  Command  (closer (E₁ ⊙⟪ σ ⟫e E₂) ξ) = Command (closer E₁ (≤-trans (_ , σ _) ξ))
                                      ∪ Command (closer E₂ (≤-trans (_ , ∙-comm σ _) ξ))
  Response (closer (E₁ ⊙⟪ σ ⟫e E₂) ξ) (inj₁ x) = Response (closer E₁ (≤-trans (_ , σ _) ξ)) x
  Response (closer (E₁ ⊙⟪ σ ⟫e E₂) ξ) (inj₂ y) = Response (closer E₂ (≤-trans (_ , ∙-comm σ _) ξ)) y
  next (closer (E₁ ⊙⟪ σ ⟫e E₂) ξ) (inj₁ x) r = next (closer E₁ (≤-trans (_ , σ _) ξ)) x r
  next (closer (E₁ ⊙⟪ σ ⟫e E₂) ξ) (inj₂ y) r = next (closer E₂ (≤-trans (_ , ∙-comm σ _) ξ)) y r
