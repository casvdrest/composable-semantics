{-# OPTIONS --safe --without-K #-}

module Fragments.NatCase where

open import Data.List
open import Data.List.Relation.Unary.All
open import Data.Product
open import Data.Nat

open import Signature
open import Canon
open import MonadicFragment

open import Function
import      Level    as L

module _ where 

  open import Fragments.Nat using (NatTyΣ ; NatVal ; `nat)
  
  open Monadic ⦃...⦄

  data NatCaseExprShape {σ} ⦃ _ : NatTyΣ ≼ σ ⦄ : withContext (μ σ) → Set where
    ncase : ∀ {Γ t} → NatCaseExprShape (Γ , t)  
  
  NatCaseExprΣ : ⦃ _ : NatTyΣ ≼ σ ⦄ → ISignature _ (withContext (μ σ))
  NatCaseExprΣ = NatCaseExprShape ▸ λ where
    (ncase {Γ} {t}) → (Γ , `nat) ∷ (Γ , t) ∷ (`nat ∷ Γ , t) ∷ []

  open IAlgebra
  interpNatCaseAlg : ⦃ _ : NatVal ⊆ V ⦄ → IAlgebra NatCaseExprΣ (withEff Reader (fold V))
  ialg interpNatCaseAlg (ncase , L.lift (fn ∷ fz ∷ fs ∷ [])) = do
    n ← fn
    case ↓ n of λ where
      zero     → fz
      (suc n′) → local (↑ n′ ∷_) fs

  frag = mkfragmentM Reader interpNatCaseAlg
