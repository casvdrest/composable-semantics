-- Algebras over functors on indexed sets
module _ where 

  Algebra : ∀ {X} → Data X → ISet X → Set
  Algebra E V = ∀[ E V ⇒ V ]

  record Effects (F : Set → Set) : Set₁ where
    field M : ∀ {X} → ISet X → Data (F X × X)

  record Monadic {F : Set → Set} (η : Effects F) : Set₁ where
    open Effects η
    
    field
      pure  : ∀ {X V} {P : F X × X → Set} {t} → P t → M V P t
      _>>=_ : ∀ {X V} {P₁ P₂ : F X × X → Set} {t₁ t₂} → M V P₁ t₁ → (P₁ t₁ → M V P₂ t₂) → M V P₂ t₂

    _>>_ : ∀ {X V} {P₁ P₂ : F X × X → Set} {t₁ t₂} → M V P₁ t₁ → M V P₂ t₂ → M V P₂ t₂
    f >> g = f >>= λ _ → g

-- Open World (O.W.) algebras
--
-- O.W. algebras are defined such that they leave open the possibility for values to be added
-- to all their dimensions. That is, we might combine two O.W. algebras that have:
-- 
-- 1. Different index types `T`
-- 2. Different value types `V`
-- 3. Different expression types `E`
--
-- Note that `E` is indeed the domain of the O.W. algebra, but that `T` and `V` are *lower bounds*!
module _ {F : Set → Set} where

  open Effects

  infix 5 _⊢_⟨_⟩↦_

  record _⊢_⟨_⟩↦_ (T : IndexType) (E : ExpressionType T F) (η : Effects F) (V : ValueType T) : Set₁ where
  
    field run : ∀ {X} {W : ISet X}
                → (ξ : T X ≤ X)
                → (μ : Mono ξ V W)
                → Algebra (⟪ E ⟫ ξ) (M η W (W ∘ proj₂))

  open _⊢_⟨_⟩↦_ public

-- O.W. Algebra composition
--
-- Composition of O.W. algebras is virtually the same as the regular coproduct of algebras,
-- the only difference being that we don't produce an algebra over a sum type, but rather we take
-- a proof-relevant separation of the expression types, that tells us if and how the expression types overlap
module _ where

  -- Separations of expression types. We define this as a pointwise application of the standard separation on `Set`
  instance expr-sep : ∀ {F T} → Rel₃ (ExpressionType T F)
  (expr-sep {F} {T} Rel₃.∙ E₁ ≣ E₂) E₃ =
    ∀ {X} {ξ : T X ≤ X} {R} {t}
    → ⟪ E₁ ⟫ ξ R t ∙ ⟪ E₂ ⟫ ξ R t ≣ ⟪ E₃ ⟫ ξ R t

  -- O.W. Algebra composition
  --
  -- Note that this definition is left-biased whenever the expression types overlap. If the algebras under composition
  -- don't define the same semantics for overlapping expression types, this operation is not commutative!
  _⊕⟨_⟩_ : ∀ {T}
             {E₁ E₂ E₃ : ExpressionType T F}
             {V : ValueType T}
             {η : Effects F}

           → T ⊢ E₁ ⟨ η ⟩↦ V
           → E₁ ∙ E₂ ≣ E₃
           → T ⊢ E₂ ⟨ η ⟩↦ V
           → T ⊢ E₃ ⟨ η ⟩↦ V
           
  run (alg₁ ⊕⟨ σe ⟩ alg₂) {W = W} ξ m e with from σe e
  ... | this  e₁   = run alg₁ {W = W} ξ m e₁
  ... | that  e₂   = run alg₂ {W = W} ξ m e₂
  ... | these e₁ _ = run alg₁ {W = W} ξ m e₁


-- Weakening operations for o.w. algebras
module _ where

  open Inverse
  open ValueType

  value-⊙≡ : ∀ {V₁ : ValueType T₁} {V₂ : ValueType T₂}
               {X : Set} {W} {t : T₁ X}
             → (σ : T₁ ∙ T₂ ≣ T₃)
             → value (V₁ ⊙⟨ σ ⟩v V₂) W (inja (σ _) t) P.≡ value V₁ W t  
  value-⊙≡ {t = t} σ with a-inv' (σ _) t
  ... | From.this a    i P.refl rewrite i = P.refl
  ... | From.these a b i P.refl rewrite i = P.refl
                                            
  ↔-resp-≡ₗ : ∀ {A B C : Set} → A P.≡ B → A ↔ C → B ↔ C
  ↔-resp-≡ₗ P.refl eqv = eqv
  
  -- Transportation between a value type, and it's weakenings
  extend⟨_⟩ : ∀ {V₁ : ValueType T₁} {V₂ : ValueType T₂}
                {X : Set} {W} {t : T₁ X}
              → (σ : T₁ ∙ T₂ ≣ T₃)
              → value V₁ W t ↔ value (V₁ ⊙⟨ σ ⟩v V₂) W (inja (σ _) t)
  extend⟨ σ ⟩ = ↔-resp-≡ₗ (value-⊙≡ σ) (I.id-↔ _)


module _ {η : Effects F} where 

  open ValueType
  open IsEquivalence ⦃...⦄
  
  -- O.W. algebra weakening
  --
  -- Simultaneously raises the lower bound on both the index set & the value type
  _▿ᴬ_ : ∀ {E : ExpressionType T₁ F} {V₁ : ValueType T₁}
         → T₁ ⊢ E ⟨ η ⟩↦ V₁ → (σt : T₁ ∙ T₂ ≣ T₃)
         → ∀ {V₂} → T₃ ⊢ E ▿ (T₂ , σt) ⟨ η ⟩↦ V₁ ⊙⟨ σt ⟩v V₂
         
  run (alg ▿ᴬ σt) ξ m e = run alg (≤-trans (_ , σt _) ξ) ((extend⟨ σt ⟩) C.∘-↔ m) e


  --O.W. algebra's respect transportation of value types
  ⟨_⟩∘_ :  ∀ {E : ExpressionType T F}
           → {V₁ V₂ : ValueType T}
           → (∀ {X} {W : X → Set} {t : T X} → value V₁ W t ↔ value V₂ W t)
           → T ⊢ E ⟨ η ⟩↦ V₁ → T ⊢ E ⟨ η ⟩↦ V₂

  run (⟨ t ⟩∘ alg) ξ m e = run alg ξ (t C.∘-↔ m) e

module _ {Δ₁ Δ₂ Δ₃ : Domains F} {η : Effects F} where

  open Domains
  infixl 10 _⊙⟨_⟩a_ 

  -- Heterogeneous algebra compositionb
  --
  -- Given two algebras, with domains `Δ₁` and `Δ₂` respectively, and a domain
  -- separation of the form Δ₁ ⊙ Δ₂ ≣ Δ₃, produce an O.W. algebra with domains `Δ₃`:
  _⊙⟨_⟩a_ :  ty Δ₁ ⊢ expr Δ₁ ⟨ η ⟩↦ val Δ₁
           → Δ₁ ⊙ Δ₂ ≣ Δ₃
           → ty Δ₂ ⊢ expr Δ₂ ⟨ η ⟩↦ val Δ₂
           → ty Δ₃ ⊢ expr Δ₃ ⟨ η ⟩↦ val Δ₃
           
  (alg₁ ⊙⟨ σt ∣ σe ∣ σv ▹ ι ⟩a alg₂) =
    (⟨ σv                ⟩∘ (alg₁ ▿ᴬ        σt))  ⊕⟨ σe ⟩                       
    (⟨ ⟨⊙⟩v-comm σt ι σv ⟩∘ (alg₂ ▿ᴬ ∙-comm σt))
