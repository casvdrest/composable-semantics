{-# OPTIONS --without-K  --safe #-}

open import Composable.Algebra

module Composable.Fragments.Nat (K : Set → Set) (η : Effects K) where 

open import Composable.Transformer
open import Composable.Data.Core
open import Composable.Data.Instances
open import Composable.Fragments.Core

open import Data.Product
open import Data.Empty
open import Data.Sum
open import Data.Unit
open import Data.Nat

open import Relation.Ternary.Construct.Sets.Union
open import Relation.Binary.PropositionalEquality

open import Function

module _ where

  open Fragment {η = η}
  open Domains
  open Open
  open ValueType

  open Union
  
  data `ℕ (X : Set) : Set where
    nat : `ℕ X

  NatExpr : ExpressionType `ℕ K
  closer NatExpr (_ , σ) R (κ , t)
         = t ≡ inja σ nat                                           -- zero
         ⊎ t ≡ inja σ nat × R (κ , inja σ nat)                      -- suc
         ⊎ t ≡ inja σ nat × R (κ , inja σ nat) × R (κ , inja σ nat) -- plus

  NatVal : ValueType `ℕ
  value NatVal R nat = ℕ

  NatDom : Domains K
  ty   NatDom = `ℕ
  expr NatDom = NatExpr
  val  NatDom = NatVal

module _ where

  open Monadic ⦃...⦄
  open Inverse renaming (f to ↑ ; f⁻¹ to ↓)

  evalNat : ⦃ Monadic η ⦄ → `ℕ ⊢ NatExpr ⟨ η ⟩↦ NatVal
  run evalNat ξ μ (inj₁ refl)                    = pure (↑ μ zero)
  run evalNat ξ μ (inj₂ (inj₁ (refl , fn)))      = do
    n ← fn
    pure (↑ μ (suc (↓ μ n)))
  run evalNat ξ μ (inj₂ (inj₂ (refl , fn , fm))) = do
    n ← fn
    m ← fm
    pure (↑ μ (↓ μ n + ↓ μ m))

  NatFrag : ⦃ Monadic η ⦄ → Fragment NatDom
  Fragment.denote NatFrag = evalNat
