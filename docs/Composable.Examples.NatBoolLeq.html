-- Fold/fixpoint 
module _ where

  open Domains
  open Inverse

  -- Type fixpoint 
  data FixN∣B : Set where
    In : ty N∣B∣Leq FixN∣B → FixN∣B

  FixN∣B-iso : FixN∣B ↔ ty N∣B∣Leq FixN∣B
  f   FixN∣B-iso (In x) = x
  f⁻¹ FixN∣B-iso = In
  cong₁ FixN∣B-iso P.refl = P.refl
  cong₂ FixN∣B-iso P.refl = P.refl
  proj₁ (inverse FixN∣B-iso) _      = P.refl
  proj₂ (inverse FixN∣B-iso) (In _) = P.refl

  -- Expression fixpoint
  data FixExpr (t : FixN∣B × FixN∣B) : Set where
    Inₑ : ⟪ expr N∣B∣Leq ⟫ (from↔ (S.sym-↔ FixN∣B-iso)) FixExpr t → FixExpr t
 
  FixExpr-iso : ∀ {t} → FixExpr t ↔ ⟪ expr N∣B∣Leq ⟫ (from↔ (S.sym-↔ FixN∣B-iso)) FixExpr t
  f FixExpr-iso (Inₑ x) = x
  f⁻¹ FixExpr-iso = Inₑ
  cong₁ FixExpr-iso P.refl = P.refl
  cong₂ FixExpr-iso P.refl = P.refl
  proj₁ (inverse FixExpr-iso) _ = P.refl
  proj₂ (inverse FixExpr-iso) (Inₑ _) = P.refl
