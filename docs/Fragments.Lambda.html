module Fragments.Lambda where

open import Signature
open import Canon
open import MonadicFragment

open import Data.Product
open import Data.Vec hiding (_>>=_ ; lookup)
open import Data.Maybe hiding (_>>=_)
open import Data.List hiding (lookup)
open import Data.List.Membership.Propositional
open import Data.List.Relation.Unary.All

open import Function
import      Level    as L

module _ where

  data FunTyShape : Set where fun : FunTyShape
  FunTyΣ = FunTyShape ▹ λ where fun → 2  

  `fun : ⦃ FunTyΣ ≼ σ ⦄ → μ σ → μ σ → μ σ
  `fun s t = inject (fun , L.lift (s ∷ t ∷ []))

  FunVal : Algebra FunTyΣ Set
  alg (FunVal) (fun , L.lift (s ∷ t ∷ [])) = s → Maybe t

  data LambdaExprShape {σ} ⦃ _ : FunTyΣ ≼ σ ⦄ : withContext (μ σ) → Set where 
    var : ∀ {Γ t} → t ∈ Γ → LambdaExprShape (Γ , t) 
    abs : ∀ {Γ s t} → LambdaExprShape (Γ , `fun s t)
    app : ∀ {Γ t} {s : μ σ} → LambdaExprShape (Γ , t)

  LambdaExprΣ : ⦃ _ : FunTyΣ ≼ σ ⦄ → ISignature _ (withContext (μ σ))
  LambdaExprΣ = LambdaExprShape ▸ λ where
    (var x)           → []
    (abs {Γ} {s} {t}) → (s ∷ Γ , t) ∷ []
    (app {Γ} {t} {s}) → (Γ , `fun s t) ∷ (Γ , s) ∷ []

  open Monadic ⦃...⦄
  open IAlgebra
  interpLambdaAlg : ⦃ _ : FunVal ⊆ V ⦄ → IAlgebra LambdaExprΣ (withEff Reader (fold V))
  ialg interpLambdaAlg (var x , L.lift []) = do
    nv ← ask
    return (lookup nv x) 
  ialg interpLambdaAlg (abs , L.lift (fx ∷ [])) = do
    nv ← ask
    n  ← remaining
    return (↑ λ x → fx n (x ∷ nv))
  ialg interpLambdaAlg (app , L.lift (ff ∷ fx ∷ [])) = do
    f ← ff
    x ← fx
    λ _ _ → ↓ f x 

  frag = mkfragmentM Reader interpLambdaAlg
