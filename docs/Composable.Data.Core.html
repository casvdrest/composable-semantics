-- The type of signature
module _ where 

  ISet : Set → Set₁
  ISet T = T → Set

  iset-resp : Respect _↔_ ISet
  Respect.coe iset-resp eqv P = P ∘ f⁻¹ 
    where open Inverse eqv

  _▸_ : Set → Set → Set₁
  I ▸ O = ISet I → ISet O

-- Some operations and properties of signatures
module _ where

  variable
    I I₁ I₂ I₃ O O₁ O₂ O₃ : Set
    Σ₁ : I₁ ▸ O₁
    Σ₂ : I₂ ▸ O₂
    Σ₃ : I₃ ▸ O₃

  _⊕_ : I ▸ O → I ▸ O → I ▸ O   
  (Σ₁ ⊕ Σ₂) R = Σ₁ R ∪ Σ₂ R

  _⊗_ : I ▸ O → I ▸ O → I ▸ O
  (Σ₁ ⊗ Σ₂) R = Σ₁ R ∩ Σ₂ R

  _⊚_ : ∀ {M} → M ▸ O → I ▸ M → I ▸ O
  Σ₁ ⊚ Σ₂ = Σ₁ ∘ Σ₂


  -- signatures respect type isomorphism in both the input and output positions

  ▸-respᴼ : Respect _↔_ (I ▸_)
  Respect.coe ▸-respᴼ eqv P R = P R ∘ f⁻¹
    where open Inverse eqv

  ▸-respᴵ : Respect _↔_ (_▸ O)
  Respect.coe ▸-respᴵ eqv P R = P (R ∘ f)
    where open Inverse eqv

-- Universes of extensible data types
--
-- Note that `Data` & friends only facilitate extension of the set of constructors! To
-- make the set of indices extensible we must use the `Open` transformer defined in
-- `Transformer.agda`
module _ where

  Data : Set → Set₁
  Data I = I ▸ I

  AugmentedData : (F : Set → Set) → Set → Set₁
  AugmentedData F I = Data (F I × I)

  injData : ∀ {F I} → Data I → AugmentedData F I
  injData D R (Γ , t) = D (λ t' → R (Γ , t')) t

  ≤→-refl : ∀ {F : Set → Set} → F ≤ F
  ≤→-refl = _ , λ _ → ∙-copy _
