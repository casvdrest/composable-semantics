-- Identity

module _ where

  open ValueType
  open Inverse

  εv : ValueType ε
  value εv R t = ⊥

  ⟨⊙⟩v-unit : ∀ {V : ValueType T}
            → V ⊙⟨ ∙-idʳ ⟩v εv ≣ V
  f              ⟨⊙⟩v-unit      = id
  f⁻¹            ⟨⊙⟩v-unit      = id
  cong₁          ⟨⊙⟩v-unit refl = refl
  cong₂          ⟨⊙⟩v-unit refl = refl
  proj₁ (inverse ⟨⊙⟩v-unit) _   = refl
  proj₂ (inverse ⟨⊙⟩v-unit) _   = refl

-- Commutativity

module _ where 

  open _←⟨_⟩→_
  open Inverse
  open ValueType


  comm-transp₁ : ∀ {V₁ V₂ X W} {t : T₃ X}
                 → (σ : T₁ ∙ T₂ ≣ T₃)
                 → (ι : Interchangeable σ V₁ V₂ X W t)
                 → value (V₂ ⊙⟨ ∙-comm σ ⟩v V₁) W t
                 → value (V₁ ⊙⟨ σ        ⟩v V₂) W t
  comm-transp₁ {t = t} σ ι x with from (σ _) t
  ... | this  _   = x 
  ... | that  _   = x 
  ... | these _ _ = f⁻¹ ι x

  comm-transp₂ : ∀ {V₁ V₂ X W} {t : T₃ X}
                 → (σ : T₁ ∙ T₂ ≣ T₃)
                 → (ι : Interchangeable σ V₁ V₂ X W t)
                 → value (V₁ ⊙⟨ σ        ⟩v V₂) W t
                 → value (V₂ ⊙⟨ ∙-comm σ ⟩v V₁) W t
  comm-transp₂ {t = t} σ ι x with from (σ _) t
  ... | this  _   = x
  ... | that  _   = x
  ... | these _ _ = f ι x 
  
  comm-transp-invˡ : ∀ {V₁ V₂ X  W} {t : T₃ X}
                     → (σ : T₁ ∙ T₂ ≣ T₃)
                     → (ι : Interchangeable σ V₁ V₂ X W t)
                     → (x : value (V₁ ⊙⟨ σ ⟩v V₂) W t)
                     → comm-transp₁ σ ι (comm-transp₂ σ ι x) ≡ x 
  comm-transp-invˡ {t = t} σ ι x  with from (σ _) t
  ... | this  _   = refl
  ... | that  _   = refl
  ... | these _ _ = proj₂ (inverse ι) _

  comm-transp-invʳ : ∀ {V₁ V₂ X  W} {t : T₃ X}
                     → (σ : T₁ ∙ T₂ ≣ T₃)
                     → (ι : Interchangeable σ V₁ V₂ X W t)
                     → (y : value (V₂ ⊙⟨ ∙-comm σ ⟩v V₁) W t)
                     → comm-transp₂ σ ι (comm-transp₁ σ ι y) ≡ y
  comm-transp-invʳ {t = t} σ ι y with from (σ _) t
  ... | this  _   = refl
  ... | that  _   = refl
  ... | these _ _ = proj₁ (inverse ι) _

  ⟨⊙⟩v-comm : ∀ {V₁ V₂ V₃}
             → (σ : T₁ ∙ T₂ ≣ T₃)
             → (ι : V₁ ←⟨ σ ⟩→ V₂)
             → V₁ ⊙⟨ σ        ⟩v V₂ ≣ V₃ 
             → V₂ ⊙⟨ ∙-comm σ ⟩v V₁ ≣ V₃
  f     (⟨⊙⟩v-comm σ ι sep) x = f sep (comm-transp₁ σ (intr ι) x)
  f⁻¹   (⟨⊙⟩v-comm σ ι sep) x = comm-transp₂ σ (intr ι) (f⁻¹ sep x)
  cong₁ (⟨⊙⟩v-comm σ ι sep) refl = refl
  cong₂ (⟨⊙⟩v-comm σ ι sep) refl = refl
  proj₁ (inverse (⟨⊙⟩v-comm σ ι sep)) x =
    P.trans (P.cong (f sep) (comm-transp-invˡ σ (intr ι) (f⁻¹ sep x)))
            (proj₁ (inverse sep) x)
  proj₂ (inverse (⟨⊙⟩v-comm σ ι sep)) y =
    P.trans (P.cong (comm-transp₂ σ (intr ι)) (proj₂ (inverse sep) (comm-transp₁ σ (intr ι) y)))
            (comm-transp-invʳ σ (intr ι) y)

module _ where 

  open Inverse
  open ValueType

  ⟨⊙⟩-idem : ∀ {V : ValueType T} → V ⊙⟨ (λ _ → ∙-copy _) ⟩v V ≣ V
  f              ⟨⊙⟩-idem      = id
  f⁻¹            ⟨⊙⟩-idem      = id
  cong₁          ⟨⊙⟩-idem refl = refl
  cong₂          ⟨⊙⟩-idem refl = refl
  proj₁ (inverse ⟨⊙⟩-idem) _   = refl
  proj₂ (inverse ⟨⊙⟩-idem) _   = refl
