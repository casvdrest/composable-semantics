module _ where

  open Fragment {η = η}
  open Domains
  open Open
  open ValueType 
  open Union

  data `Maybe (X : Set) : Set where
    maybe : X → `Maybe X

  MaybeExpr : ExpressionType `Maybe K
  closer MaybeExpr (_ , σ) R (κ , t)
         = (∃ λ t' → t ≡ inja σ (maybe t'))                                            -- nothing
         ⊎ (∃ λ t' → t ≡ inja σ (maybe t') × R (κ , t'))                               -- just
         ⊎ ∃ λ t' → t ≡ inja σ (maybe t') × R (κ , inja σ (maybe (inja σ (maybe t')))) -- join

  MaybeVal : ValueType `Maybe
  value MaybeVal R (maybe t) = Maybe (R t)

  MaybeDom : Domains K
  ty   MaybeDom = `Maybe
  expr MaybeDom = MaybeExpr
  val  MaybeDom = MaybeVal

module _ where
 
  open Monadic ⦃...⦄
  open Inverse renaming (f to ↑ ; f⁻¹ to ↓)

  join : ∀ {A : Set} → Maybe (Maybe A) → Maybe A
  join nothing         = nothing
  join (just nothing)  = nothing
  join (just (just x)) = just x

  mmap : ∀ {A B : Set} → (A → B) → Maybe A → Maybe B
  mmap f nothing = nothing
  mmap f (just x) = just (f x)

  evalMaybe : ⦃ Monadic η ⦄ → `Maybe ⊢ MaybeExpr ⟨ η ⟩↦ MaybeVal
  run evalMaybe ξ μ (inj₁ (_ , refl)) = pure (↑ μ nothing)
  run evalMaybe ξ μ (inj₂ (inj₁ (_ , refl , fx))) = do
    x ← fx
    pure (↑ μ (just x))
  run evalMaybe ξ μ (inj₂ (inj₂ (_ , refl , fm))) = do
    m ← fm
    let m' = ↓ μ m
    pure (↑ μ (join (mmap (↓ μ) (↓ μ m))))

  MaybeFrag : ⦃ Monadic η ⦄ → Fragment MaybeDom
  Fragment.denote MaybeFrag = evalMaybe
