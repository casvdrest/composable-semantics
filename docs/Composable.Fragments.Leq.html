open import Composable.Algebra

module Composable.Fragments.Leq (K : Set → Set) (η : Effects K) where 

open import Composable.Transformer
open import Composable.Data.Core
open import Composable.Data.Instances
open import Composable.Fragments.Core

  -- Resuse types and values from Nat/Bool
open import Composable.Fragments.Nat  K η using (`ℕ ; NatVal ; nat)
open import Composable.Fragments.Bool K η using (`Bool ; BoolVal ; bool)

open import Data.Product
open import Data.Empty
open import Data.Sum hiding (swap)
open import Data.Unit
open import Data.Bool
open import Data.Nat

open import Relation.Ternary.Construct.Sets
open import Relation.Ternary.Construct.Function
open import Relation.Ternary.Construct.Sets.Union
open import Relation.Ternary.Structures.Syntax hiding (_∣_)

open import Relation.Binary.PropositionalEquality
open import Relation.Unary

open import Function

module _ where

  open Fragment {η = η}
  open Domains
  open Open

  open Union

  LeqExpr : ExpressionType (`ℕ ∪ `Bool) K
  closer LeqExpr (_ , σ) R (κ , t)
         = t ≡ inja σ (inj₂ bool) × R (κ , inja σ (inj₁ nat)) × R (κ , inja σ (inj₁ nat))

  LeqDom : Domains K
  ty   LeqDom = `ℕ ∪ `Bool
  expr LeqDom = LeqExpr
  val  LeqDom = NatVal ⊙⟨ (λ _ → ∙-disjoint) ⟩v BoolVal

module _ where
 
  open Monadic ⦃...⦄
  open Inverse renaming (f to ↑ ; f⁻¹ to ↓)

  comp : (n m : ℕ) → Bool
  comp zero    m       = true
  comp (suc n) zero    = false
  comp (suc n) (suc m) = comp n m

  evalLeq : ⦃ Monadic η ⦄ → `ℕ ∪ `Bool ⊢ LeqExpr ⟨ η ⟩↦ (NatVal ⊙⟨ (λ _ → ∙-disjoint) ⟩v BoolVal)
  run evalLeq ξ μ (refl , fn , fm) = do
    n ← fn
    m ← fm
    pure (↑ μ (comp (↓ μ n) (↓ μ m)))

  LeqFrag : ⦃ Monadic η ⦄ → Fragment LeqDom
  Fragment.denote LeqFrag = evalLeq
