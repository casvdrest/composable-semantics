module _ where

  record Fragment {η} (Δ : Domains F) : Set₁ where
    field denote : let open Domains Δ in
                    ty ⊢ expr ⟨ η ⟩↦ val   


module _ {η : Effects F} {Δ₁ Δ₂ Δ₃ : Domains F} where

  open Fragment {F} {η}

  infixr 10 _⊙⟨_⟩f_

  _⊙⟨_⟩f_ :   (Φ₁ : Fragment Δ₁)
            → Δ₁ ⊙ Δ₂ ≣ Δ₃
            → (Φ₂ : Fragment Δ₂)
              ------------------
            → Fragment Δ₃
            
  denote (Φ₁ ⊙⟨ σ ⟩f Φ₂) = denote Φ₁ ⊙⟨ σ ⟩a denote Φ₂

module _ where

  open Domains
  open Inverse
  open ValueType

  from↔ : ∀ {A B} → A ↔ B → A ≤ B
  proj₁           (from↔ eqv)    = ⊥
  inja     (proj₂ (from↔ eqv))   = f eqv
  injb     (proj₂ (from↔ eqv))   = λ()
  from     (proj₂ (from↔ eqv)) x = this (f⁻¹ eqv x)
  a-inv    (proj₂ (from↔ eqv)) a = proj₂ (inverse eqv) a
  b-inv    (proj₂ (from↔ eqv)) ()
  from-inv (proj₂ (from↔ eqv))   = proj₁ (inverse eqv)

  record HasFixpoint (Δ : Domains F) : Set₁ where
    field μty       : Set 
          μval      : μty → Set
          μexpr     : (F μty × μty) → Set

          μty-iso   : μty ↔ ty Δ μty

          μval-iso  : ∀ {t}
                      → μval (f⁻¹ μty-iso t)
                      ↔ value (val Δ) μval t

          μexpr-iso : ∀ {κ t}
                      → μexpr (κ , t)
                      ↔ ⟪ expr Δ ⟫ (from↔ (S.sym-↔ μty-iso)) μexpr (κ , t)


module _ {η : Effects F} {Δ : Domains F} where

  open Domains
  open Inverse
  open Effects
  open Fragment {F} {η}
  open HasFixpoint ⦃...⦄

  record Foldable (Δ : Domains F) ⦃ _ : HasFixpoint Δ ⦄ : Set₁ where
    field foldexpr : ∀ {V : ISet (F μty × μty)} {t}
                     → (⟪ expr Δ ⟫ (from↔ (S.sym-↔ μty-iso)) V t → V t)
                     → μexpr t → V t 
    
  open Foldable ⦃...⦄

  -- A fragment is 'interpretable' if:
  -- * it's domains have a fixpoint
  -- * it's expression type is foldable
  interp : ∀ (Φ : Fragment {F} {η} Δ)
           → ⦃ _ : HasFixpoint Δ ⦄
           → ⦃ Foldable Δ ⦄
           → ∀ {κ : F μty} {t : μty} → μexpr (κ , t) → M η μval (μval ∘ proj₂) (κ , t)
           
  interp Φ x =
    foldexpr (run (denote Φ) (from↔ (S.sym-↔ μty-iso)) (S.sym-↔ μval-iso)) x
